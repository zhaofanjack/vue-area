## API

```js
    <vue-area :props-show="show" :props-result="result" v-on:result="areaResult"></vue-area>
```

```show``` 显示隐藏组件

```result```  省市区数据

## Usage

```js
'use strict'
import Vue from 'vue'
import vueArea from 'vue-area'

var app = new Vue({
  el: '#app',
  components: {
    vueArea
  },
  methods: {
    areaResult: function(show, result){
        this.show = show
        this.result = result
    }
  },
  data: {
    result: null,
    show: false
  }
})
```

```html
<vue-area :props-show="show" :props-result="result" v-on:result="areaResult"></vue-area>
```
